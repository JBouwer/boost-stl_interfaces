         typename Reference = ValueType &, typename Pointer = ValueType *,

         typename E = std::enable_if_t<            std::is_class<Derived>::value &&            std::is_same<Derived, std::remove_cv_t<Derived>>::value> > 
struct iterator_interface {

  typedef unspecified     iterator_category;

  typedef DifferenceType  difference_type;      
  typedef Derived         derived_iterator_type;

  // public member functions

Description

A CRTP template that one may derive from to make defining iterators easier.
