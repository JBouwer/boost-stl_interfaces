BOOST_ITERATOR_FACADE_STATIC_ASSERT_CONCEPT(iter, concept_name)
BOOST_ITERATOR_FACADE_STATIC_ASSERT_ITERATOR_TRAITS(iter, category, concept, value_type, reference, pointer, difference_type)
namespace boost {
  namespace iterator_facade {
    struct access;
    template<typename Derived, typename IteratorConcept, typename ValueType, 
             typename Reference = ValueType &, typename Pointer = ValueType *, 
             typename DifferenceType = std::ptrdiff_t> 
      struct iterator_facade;

    template<typename Derived, typename ValueType, typename Reference, 
             typename Pointer, typename DifferenceType> 
      struct iterator_facade<Derived, std::bidirectional_iterator_tag, ValueType, Reference, Pointer, DifferenceType>;
    template<typename Derived, typename ValueType, typename Reference, 
             typename Pointer, typename DifferenceType> 
      struct iterator_facade<Derived, std::contiguous_iterator_tag, ValueType, Reference, Pointer, DifferenceType>;
    template<typename Derived, typename ValueType, typename Reference, 
             typename Pointer, typename DifferenceType> 
      struct iterator_facade<Derived, std::forward_iterator_tag, ValueType, Reference, Pointer, DifferenceType>;
    template<typename Derived, typename ValueType, typename Reference, 
             typename Pointer, typename DifferenceType> 
      struct iterator_facade<Derived, std::input_iterator_tag, ValueType, Reference, Pointer, DifferenceType>;
    template<typename Derived, typename ValueType, typename Reference, 
             typename Pointer, typename DifferenceType> 
      struct iterator_facade<Derived, std::output_iterator_tag, ValueType, Reference, Pointer, DifferenceType>;
    template<typename Derived, typename ValueType, typename Reference, 
             typename Pointer, typename DifferenceType> 
      struct iterator_facade<Derived, std::random_access_iterator_tag, ValueType, Reference, Pointer, DifferenceType>;

    template<typename T> struct proxy_arrow_result;
    typedef iterator_facade< Derived, IteratorConcept, ValueType, Reference, proxy_arrow_result< Reference >, DifferenceType > proxy_iterator_facade;
  }
}
