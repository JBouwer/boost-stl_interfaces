namespace boost {
  namespace stl_interfaces {
    template<typename Derived, bool Contiguous = discontiguous> 
      struct container_interface;
    template<typename ContainerInterface> 
      constexpr auto swap(ContainerInterface &, ContainerInterface &);
    template<typename ContainerInterface> 
      constexpr auto operator<=(ContainerInterface, ContainerInterface);
    template<typename ContainerInterface> 
      constexpr auto operator>(ContainerInterface, ContainerInterface);
    template<typename ContainerInterface> 
      constexpr auto operator>=(ContainerInterface, ContainerInterface);
  }
